Shortest path in Graph
For shortest path in unweight graph use BFS.

https://leetcode.com/problems/nearest-exit-from-entrance-in-maze/

class Solution {
public:
    int nearestExit(vector<vector<char>>& maze, vector<int>& entrance) {
        int dist=0;
        vector<vector<int>>move={{0,1},{1,0},{0,-1},{-1,0}};
        queue<pair<int,int>>q;
        q.push(make_pair(entrance[0],entrance[1]));
        maze[entrance[0]][entrance[1]]='+';       // marking it so that starting pointed dont get visited again
        while(!q.empty()){
            
            int qConstSize=q.size();             // since q size is changing in the loop and want the initial size of queue
            for(int k=0;k<qConstSize;++k){
            
            int x=q.front().first; 
            int y=q.front().second;
            q.pop();
                
                for(int i=0;i<4;++i){
                    
                    int X=x+move[i][0];
                    int Y=y+move[i][1];
                    
                    if( X<0 || Y<0 || X>maze.size()-1 || Y>maze[0].size()-1){
                        continue;   
                    }

                    if(maze[X][Y]=='.'){              // checking if it can be visited

                        maze[X][Y]='+';               // marking it so that we should not visit it again
                        q.push(make_pair(X,Y));

                        if( X==0 || Y==0 || X==maze.size()-1 || Y==maze[0].size()-1){
                            return dist+1;
                        }
                    }
                }
            }
            ++dist;
        }
        return -1;
    }
};
