0-1 KNAPSACK

Recursion ( we can remove if else by using adding base condition for W<0 )

int knapSack(int W, int wt[], int val[], int n){
    if (n == 0 || W == 0) return 0;

    if (wt[n - 1] > W)
        return knapSack(W, wt, val, n - 1);
    else
        return max(val[n - 1] + knapSack(W - wt[n - 1], wt, val, n - 1),knapSack(W, wt, val, n - 1));
}

Top down ( we can remove if else by using adding base condition for W<0 )

int knapSackRec(int W, int wt[],int val[], int i,vector<vector<int>>&dp){
    if (i < 0) return 0;
    if (dp[i][W] != -1) return dp[i][W];
 
    if (wt[i] > W) {
        return dp[i][W] = knapSackRec(W, wt,val, i - 1,dp);   
    }else {
        return dp[i][W] = max(knapSackRec(W - wt[i],wt, val,i - 1, dp)+val[i],knapSackRec(W, wt, val,i - 1, dp));                               
    }
	
}

int knapSack(int W, int wt[], int val[], int n)
{
    int i, w;
    int K[n + 1][W + 1];
 
    for(i = 0; i <= n; i++)
    {
        for(w = 0; w <= W; w++)
        {
            if (i == 0 || w == 0) K[i][w] = 0;
            else if (wt[i - 1] <= w) K[i][w] = max(val[i - 1] + K[i - 1][w - wt[i - 1]], K[i - 1][w]);
            else  K[i][w] = K[i - 1][w];
        }
    }
    return K[n][W];
}

0-1 KNAPSACK IN O(n) SPACE

int KnapSack(int val[], int wt[], int n, int W)
{
    int dp[W+1];                                           // array to store final result dp[i] stores the profit with KnapSack capacity "i"
    memset(dp, 0, sizeof(dp));
 
    for(int i=0; i < n; i++){                              // iterate through all items
        for(int j=W; j>=wt[i]; j--){                       //traverse dp array from right to left
            dp[j] = max(dp[j] , val[i] + dp[j-wt[i]]);     // finds out maximum of  dp[j](excluding ith element value) and val[i] + dp[j-wt[i]] (including ith element value and the profit with "KnapSack capacity - ith element weight") */   
		}
	}
	return dp[W];
}

UNBOUNDED KNAPSACK

Top Down similar to bounded with some modification in choices.

Bottom Up in O(n)

int unboundedKnapsack(int W, int n, int val[], int wt[]){
    int dp[W+1];
    memset(dp, 0, sizeof dp);

    for (int i=0; i<=W; i++)
      for (int j=0; j<n; j++)
         if (wt[j] <= i) dp[i] = max(dp[i], dp[i-wt[j]] + val[j]);
 
    return dp[W];
}

class Solution {
public:
    int numSquares(int n) {
        int x=sqrt(n);
        if(x==sqrt(n)) return 1;
        return solve(x,n,0);
    }
    
    int solve(int n,int target,int sum){
        if(sum==target) return 0;
        
        for(int i=n;i>=1;--i){
            int square=i*i;
            if(sum+square>target) return solve(n-1,target,sum);
            else return min(solve(n,target,sum+square)+1,solve(n-1,target,sum));
        }
        return INT_MAX;
    }
};
